代码示例

// 基本用法
copyToClipboard('要复制的文本');

// 带选项的用法
copyToClipboard('要复制的文本', {
    silent: true  // 静默执行，不显示提示
}).then(success => {
    if (success) {
        console.log('复制成功！');
    } else {
        console.log('复制失败');
    }
});
